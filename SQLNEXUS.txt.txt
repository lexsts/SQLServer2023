SELECT SESSION_ID,statement_start_offset,TRANSACTION_BEGIN_TIME,MAX(RUNTIME) TRANSACTION_END_TIME, LAST_REQUEST_START_TIME,tran_name 
FROM TBL_REQUESTS WHERE SESSION_ID=370 AND LAST_REQUEST_START_TIME='2022-01-06 04:22:03.287' 
AND TRANSACTION_BEGIN_TIME IS NOT NULL AND TRAN_NAME<>'sqlsource_transform'
GROUP BY SESSION_ID,statement_start_offset,TRANSACTION_BEGIN_TIME,LAST_REQUEST_START_TIME,TRAN_NAME
order by MAX(RUNTIME)

SELECT SESSION_ID,RUNTIME,TRANSACTION_BEGIN_TIME,WAIT_TYPE,WAIT_DURATION_MS,LAST_REQUEST_START_TIME,tran_name 
FROM TBL_REQUESTS WHERE SESSION_ID=370 and WAIT_DURATION_MS>10000 order by RUNTIME
